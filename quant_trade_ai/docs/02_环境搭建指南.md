### 4.4 函数定义

```python
def calculate_profit(buy_price, sell_price, shares):
    """
    计算交易利润
    
    参数:
        buy_price: 买入价格
        sell_price: 卖出价格
        shares: 股数
    
    返回:
        利润金额
    """
    profit = (sell_price - buy_price) * shares
    return profit

# 调用函数
my_profit = calculate_profit(100, 120, 10)
print(f"利润: ${my_profit}")  # 输出: 利润: $200
```

### 4.5 类和对象

```python
class TradingAccount:
    """交易账户类"""
    
    def __init__(self, initial_balance):
        """构造函数，创建账户时调用"""
        self.balance = initial_balance  # 账户余额
        self.shares = 0                 # 持有股数
    
    def buy(self, price, quantity):
        """买入股票"""
        cost = price * quantity
        if cost <= self.balance:
            self.balance -= cost
            self.shares += quantity
            print(f"买入 {quantity} 股，花费 ${cost}")
    
    def sell(self, price, quantity):
        """卖出股票"""
        if quantity <= self.shares:
            revenue = price * quantity
            self.balance += revenue
            self.shares -= quantity
            print(f"卖出 {quantity} 股，收入 ${revenue}")

# 使用类
account = TradingAccount(10000)  # 创建账户，初始资金1万
account.buy(100, 50)             # 以100元买入50股
account.sell(120, 50)            # 以120元卖出50股
```

## 5. NumPy 基础

NumPy 是科学计算的基础库，用于高效处理数组。

```python
import numpy as np

# 创建数组
prices = np.array([100, 105, 103, 108, 110])

# 基本运算
mean_price = np.mean(prices)     # 平均值: 105.2

returns = np.diff(prices) / prices[:-1]

random_returns = np.random.normal(0, 0.02, 100)  # 100个正态分布随机数
```

## 6. Pandas 基础

Pandas 用于处理表格数据，非常适合股票数据分析。

```python
import pandas as pd

# 创建 DataFrame
data = {
    'Date': ['2024-01-01', '2024-01-02', '2024-01-03'],
    'Open': [100, 105, 103],
    'Close': [105, 103, 108],
    'Volume': [1000000, 1200000, 900000]
}
df = pd.DataFrame(data)

# 查看数据
print(df.head())        # 前5行
print(df.describe())    # 统计摘要

# 计算指标
df['Return'] = df['Close'].pct_change()  # 收益率
df['MA5'] = df['Close'].rolling(5).mean() # 5日均线

# 筛选数据
high_volume = df[df['Volume'] > 1000000]  # 成交量大于100万的日期
```

## 7. PyTorch 基础

PyTorch 是深度学习框架，用于构建神经网络。

```python
import torch
import torch.nn as nn

# 张量 (Tensor) - PyTorch 的基本数据结构
x = torch.tensor([1.0, 2.0, 3.0])
y = torch.tensor([4.0, 5.0, 6.0])
z = x + y  # 逐元素相加

# 创建简单的神经网络
class SimpleNet(nn.Module):
    def __init__(self, input_size, output_size):
        super().__init__()
        self.layer = nn.Linear(input_size, output_size)
    
    def forward(self, x):
        return self.layer(x)

# 创建网络实例
net = SimpleNet(input_size=10, output_size=3)

# 前向传播
input_data = torch.randn(1, 10)  # 随机输入
output = net(input_data)         # 获得输出
```
