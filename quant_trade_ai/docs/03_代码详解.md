## 1. 交易环境 (trading_env.py)

交易环境是强化学习的核心组件，模拟真实的股票交易过程。

### 1.1 整体结构

```
TradingEnvironment 类

├── _calculate_indicators()  # 计算技术指标
├── reset()           # 重置环境
├── _get_observation()  # 获取当前状态
├── step()            # 执行一步交易
```

### 1.2 代码逐行解释

```python
"""
交易环境 - 模拟股票交易市场
基于 Gymnasium 接口实现

Gymnasium 是 OpenAI Gym 的继承者，提供标准的强化学习环境接口
所有环境都需要实现 reset() 和 step() 方法
"""

import numpy as np          # 数值计算库
import pandas as pd         # 数据处理库
import gymnasium as gym     # 强化学习环境标准接口
from gymnasium import spaces  # 定义状态/动作空间
import ta                   # 技术指标计算库


class TradingEnvironment(gym.Env):
    """
    股票交易强化学习环境
    
    继承自 gym.Env，需要实现以下核心方法：
    - reset(): 重置环境到初始状态
    - step(action): 执行动作并返回结果
    
    状态空间: 技术指标 + 持仓状态 + 账户信息
    动作空间: 0-持有, 1-买入, 2-卖出
    """
    
    # 元数据，定义渲染模式
    metadata = {'render_modes': ['human']}
    
    def __init__(self, df: pd.DataFrame, initial_balance: float = 100000, 
                 transaction_fee: float = 0.001, window_size: int = 20):
        """
        初始化交易环境
        
        参数说明:
            df: 包含OHLCV数据的DataFrame
                - Open: 开盘价
                - High: 最高价
                - Low: 最低价
                - Close: 收盘价
                - Volume: 成交量
            initial_balance: 初始资金，默认10万
            transaction_fee: 交易手续费比例，默认0.1%
            window_size: 观察窗口大小，计算技术指标需要的历史数据长度
        """
        super().__init__()  # 调用父类构造函数
        
        # 保存参数
        self.df = df.copy()  # 复制数据，避免修改原始数据
        self.initial_balance = initial_balance
        self.transaction_fee = transaction_fee
        self.window_size = window_size
        
        # 计算技术指标（在原始数据上添加新列）
        self._calculate_indicators()
        
        # 定义使用的特征列
        # 这些是智能体能"看到"的市场信息
        self.feature_columns = [
            'returns',      # 收益率
            'rsi',          # 相对强弱指数
            'macd',         # MACD值
            'macd_signal',  # MACD信号线
            'macd_diff',    # MACD柱状图
            'bb_high',      # 布林带上轨距离
            'bb_low',       # 布林带下轨距离
            'bb_mid',       # 布林带中轨距离
            'sma_20',       # 20日简单移动平均
            'ema_20',       # 20日指数移动平均
            'volume_sma',   # 成交量相对均值
            'atr'           # 平均真实波幅
        ]
        
        # ============================================
        # 定义状态空间 (Observation Space)
        # ============================================
        # 状态包括：技术指标 + 持仓比例 + 未实现盈亏
        n_features = len(self.feature_columns) + 2
        
        # Box 表示连续空间，范围从 -∞ 到 +∞
        self.observation_space = spaces.Box(
            low=-np.inf,    # 最小值
            high=np.inf,    # 最大值
            shape=(n_features,),  # 形状：一维向量
            dtype=np.float32      # 数据类型
        )
        
        # ============================================
        # 定义动作空间 (Action Space)
        # ============================================
        # Discrete(3) 表示离散空间，有3个选项：0, 1, 2
        # 0 = 持有 (Hold)
        # 1 = 买入 (Buy)
        # 2 = 卖出 (Sell)
        self.action_space = spaces.Discrete(3)
        
        # 重置环境到初始状态
        self.reset()
    
    def _calculate_indicators(self):
        """
        计算技术指标
        
        技术指标是根据历史价格计算出的数值，
        帮助判断市场趋势和买卖时机
        """
        df = self.df
        
        # ============================================
        # 1. 收益率 (Returns)
        # ============================================
        # pct_change() 计算相邻两天的百分比变化
        # 例如：今天100，明天105，收益率 = 5%
        df['returns'] = df['Close'].pct_change()
        
        # ============================================
        # 2. RSI - 相对强弱指数 (Relative Strength Index)
        # ============================================
        # RSI 衡量价格上涨和下跌的相对强度
        # 范围：0-100
        # > 70: 超买（可能要跌）
        # < 30: 超卖（可能要涨）
        # 除以100进行归一化，使范围变为0-1
        df['rsi'] = ta.momentum.RSIIndicator(
            df['Close'], 
            window=14  # 使用14天数据计算
        ).rsi() / 100
        
        # ============================================
        # 3. MACD - 移动平均收敛散度
        # ============================================
        # MACD 由三部分组成：
        # - MACD线：快速EMA(12) - 慢速EMA(26)
        # - 信号线：MACD线的9日EMA
        # - 柱状图：MACD线 - 信号线
        macd = ta.trend.MACD(df['Close'])
        df['macd'] = macd.macd() / df['Close']  # 归一化
        df['macd_signal'] = macd.macd_signal() / df['Close']
        df['macd_diff'] = macd.macd_diff() / df['Close']
        
        # ============================================
        # 4. Bollinger Bands - 布林带
        # ============================================
        # 布林带由三条线组成：
        # - 中轨：20日移动平均
        # - 上轨：中轨 + 2倍标准差
        # - 下轨：中轨 - 2倍标准差
        # 价格触及上轨可能回调，触及下轨可能反弹
        bb = ta.volatility.BollingerBands(df['Close'], window=20)
        df['bb_high'] = (bb.bollinger_hband() - df['Close']) / df['Close']
        df['bb_low'] = (df['Close'] - bb.bollinger_lband()) / df['Close']
        df['bb_mid'] = (bb.bollinger_mavg() - df['Close']) / df['Close']
        
        # ============================================
        # 5. 移动平均线 (Moving Averages)
        # ============================================
        # SMA: 简单移动平均，每天权重相同
        # EMA: 指数移动平均，近期数据权重更大
        df['sma_20'] = (ta.trend.SMAIndicator(
            df['Close'], window=20
        ).sma_indicator() - df['Close']) / df['Close']
        
        df['ema_20'] = (ta.trend.EMAIndicator(
            df['Close'], window=20
        ).ema_indicator() - df['Close']) / df['Close']
        
        # ============================================
        # 6. 成交量指标
        # ============================================
        # 当前成交量相对于20日平均成交量的比例
        # > 1: 成交量放大（可能有重要变化）
        # < 1: 成交量萎缩
        df['volume_sma'] = df['Volume'] / df['Volume'].rolling(window=20).mean() - 1
        
        # ============================================
        # 7. ATR - 平均真实波幅 (Average True Range)
        # ============================================
        # 衡量价格波动的剧烈程度
        # ATR越大，价格波动越剧烈
        df['atr'] = ta.volatility.AverageTrueRange(
            df['High'], df['Low'], df['Close']
        ).average_true_range() / df['Close']
        
        # 填充缺失值（前几天可能因为计算窗口不足而产生NaN）
        df.fillna(0, inplace=True)
        
        self.df = df
    
    def reset(self, seed=None, options=None):
        """
        重置环境到初始状态
        
        每个交易回合开始时调用，重新开始模拟交易
        
        返回:
            observation: 初始状态观测
            info: 额外信息字典
        """
        super().reset(seed=seed)  # 设置随机种子
        
        # 从窗口大小位置开始，因为前面的数据用于计算技术指标
        self.current_step = self.window_size
        
        # 重置账户状态
        self.balance = self.initial_balance  # 现金余额
        self.shares_held = 0                  # 持有股数
        self.total_shares_sold = 0            # 总卖出股数
        self.total_sales_value = 0            # 总卖出金额
        self.cost_basis = 0                   # 持仓成本价
        self.total_profit = 0                 # 累计已实现利润
        self.trades = []                      # 交易记录列表
        
        # 返回初始观测和空的info字典
        return self._get_observation(), {}
    
    def _get_observation(self):
        """
        获取当前状态观测
        
        状态包括：
        1. 技术指标特征（12个）
        2. 持仓比例（当前持仓占总资产的比例）
        3. 未实现盈亏比例
        
        返回:
            obs: 状态向量，shape=(14,)
        """
        # 获取当前时刻的技术指标特征
        features = self.df[self.feature_columns].iloc[self.current_step].values
        
        # 计算持仓比例
        current_price = self.df['Close'].iloc[self.current_step]
        total_value = self.balance + self.shares_held * current_price
        
        # 持仓比例 = 股票市值 / 总资产
        if total_value > 0:
            position_ratio = (self.shares_held * current_price) / total_value
        else:
            position_ratio = 0
        
        # 计算未实现盈亏比例
        # 未实现盈亏 = (当前价 - 成本价) / 成本价
        unrealized_pnl = 0
        if self.shares_held > 0 and self.cost_basis > 0:
            unrealized_pnl = (current_price - self.cost_basis) / self.cost_basis
        
        # 拼接所有特征
        obs = np.concatenate([features, [position_ratio, unrealized_pnl]])
        
        return obs.astype(np.float32)
    
    def step(self, action):
        """
        执行交易动作
        
        这是环境的核心方法，执行一步模拟：
        1. 根据动作执行买入/卖出/持有
        2. 时间前进一步
        3. 计算奖励
        4. 返回新状态
        
        参数:
            action: 动作，0=持有, 1=买入, 2=卖出
        
        返回:
            observation: 新的状态观测
            reward: 奖励值
            terminated: 是否结束（到达数据末尾）
            truncated: 是否截断
            info: 额外信息
        """
        current_price = self.df['Close'].iloc[self.current_step]
        
        # 记录交易前的总资产（用于计算奖励）
        prev_total_value = self.balance + self.shares_held * current_price
        
        # ============================================
        # 执行动作
        # ============================================
        if action == 1:  # 买入
            self._buy(current_price)
        elif action == 2:  # 卖出
            self._sell(current_price)
        # action == 0 时不执行任何操作（持有）
        
        # 时间前进一步
        self.current_step += 1
        
        # 检查是否到达数据末尾
        terminated = self.current_step >= len(self.df) - 1
        truncated = False
        
        # ============================================
        # 计算奖励
        # ============================================
        # 获取新价格
        new_price = self.df['Close'].iloc[self.current_step]
        current_total_value = self.balance + self.shares_held * new_price
        
        # 奖励 = 资产变化的百分比 × 100
        # 乘以100是为了让奖励值更明显
        reward = (current_total_value - prev_total_value) / prev_total_value * 100
        
        # 风险惩罚：大幅亏损时额外惩罚
        # 这鼓励智能体避免高风险操作
        if reward < -2:
            reward *= 1.5
        
        # 返回额外信息
        info = {
            'total_value': current_total_value,  # 总资产
            'balance': self.balance,              # 现金余额
            'shares_held': self.shares_held,      # 持有股数
            'current_price': new_price,           # 当前价格
            'total_profit': current_total_value - self.initial_balance  # 总盈亏
        }
        
        return self._get_observation(), reward, terminated, truncated, info
    
    def _buy(self, price):
        """
        买入股票
        
        策略：使用50%的可用资金买入
        这样做可以：
        1. 保留部分现金应对下跌
        2. 允许多次加仓
        
        参数:
            price: 当前股价
        """
        if self.balance > 0:
            # 计算可买入金额（50%现金）
            buy_amount = self.balance * 0.5
            
            # 计算可买入股数（考虑手续费）
            # 总成本 = 股价 × 股数 × (1 + 手续费率)
            shares_to_buy = int(buy_amount / (price * (1 + self.transaction_fee)))
            
            if shares_to_buy > 0:
                # 计算实际花费
                cost = shares_to_buy * price * (1 + self.transaction_fee)
                self.balance -= cost
                
                # 更新成本基础（加权平均）
                if self.shares_held == 0:
                    # 首次买入，成本就是当前价格
                    self.cost_basis = price
                else:
                    # 加仓，计算加权平均成本
                    self.cost_basis = (
                        self.cost_basis * self.shares_held + price * shares_to_buy
                    ) / (self.shares_held + shares_to_buy)
                
                self.shares_held += shares_to_buy
                
                # 记录交易
                self.trades.append(('BUY', self.current_step, price, shares_to_buy))
    
    def _sell(self, price):
        """
        卖出股票
        
        策略：卖出50%的持仓
        这样做可以：
        1. 部分获利了结
        2. 保留部分仓位以防继续上涨
        
        参数:
            price: 当前股价
        """
        if self.shares_held > 0:
            # 卖出50%持仓
            shares_to_sell = self.shares_held // 2
            
            # 如果只剩1股，全部卖出
            if shares_to_sell == 0:
                shares_to_sell = self.shares_held
            
            # 计算卖出收入（扣除手续费）
            sell_value = shares_to_sell * price * (1 - self.transaction_fee)
            self.balance += sell_value
            self.shares_held -= shares_to_sell
            
            # 计算已实现盈亏
            profit = (price - self.cost_basis) * shares_to_sell
            self.total_profit += profit
            
            # 更新统计数据
            self.total_shares_sold += shares_to_sell
            self.total_sales_value += sell_value
            
            # 记录交易
            self.trades.append(('SELL', self.current_step, price, shares_to_sell))
            
            # 如果全部卖出，重置成本基础
            if self.shares_held == 0:
                self.cost_basis = 0
    
    def render(self, mode='human'):
        """
        渲染当前状态（打印到控制台）
        
        显示当前的交易状态信息
        """
        current_price = self.df['Close'].iloc[self.current_step]
        total_value = self.balance + self.shares_held * current_price
        profit_pct = (total_value - self.initial_balance) / self.initial_balance * 100
        
        print(f"Step: {self.current_step}")
        print(f"Price: ${current_price:.2f}")
        print(f"Balance: ${self.balance:.2f}")
        print(f"Shares: {self.shares_held}")
        print(f"Total Value: ${total_value:.2f}")
        print(f"Profit: {profit_pct:.2f}%")
        print("-" * 40)
```

### 2.2 代码逐行解释

```python
"""
DQN 交易智能体
基于 Deep Q-Network 的强化学习交易策略

DQN 的核心思想：
用神经网络来近似 Q(s, a) 函数，
Q值表示在状态s下执行动作a能获得的预期累计奖励
"""

import numpy as np
import torch
import torch.nn as nn        # 神经网络模块
import torch.optim as optim  # 优化器
import random
from collections import deque, namedtuple


# ============================================
# 经验元组定义
# ============================================
# namedtuple 创建一个命名元组，类似于简单的类
# 用于存储一次交互的所有信息
Experience = namedtuple('Experience', [
    'state',      # 当前状态
    'action',     # 执行的动作
    'reward',     # 获得的奖励
    'next_state', # 下一个状态
    'done'        # 是否结束
])


class QNetwork(nn.Module):
    """
    Q值网络
    
    输入: 状态特征向量 (14维)
    输出: 每个动作的Q值 (3维，对应买入/持有/卖出)
    
    网络结构:
    输入(14) -> 隐藏层1(128) -> 隐藏层2(64) -> 隐藏层3(32) -> 输出(3)
    """
    
    def __init__(self, state_size: int, action_size: int, 
                 hidden_sizes: list = [128, 64, 32]):
        """
        初始化Q网络
        
        参数:
            state_size: 状态维度（输入大小）
            action_size: 动作数量（输出大小）
            hidden_sizes: 隐藏层大小列表
        """
        super(QNetwork, self).__init__()
        
        layers = []  # 存储所有层
        prev_size = state_size
        
        # 构建隐藏层
        for hidden_size in hidden_sizes:
            # Linear: 全连接层，y = Wx + b
            layers.append(nn.Linear(prev_size, hidden_size))
            
            # ReLU: 激活函数，f(x) = max(0, x)
            # 引入非线性，让网络能学习复杂模式
            layers.append(nn.ReLU())
            
            # Dropout: 随机丢弃20%的神经元
            # 防止过拟合，提高泛化能力
            layers.append(nn.Dropout(0.2))
            
            prev_size = hidden_size
        
        # 输出层（不需要激活函数，因为Q值可以是任意实数）
        layers.append(nn.Linear(prev_size, action_size))
        
        # Sequential: 将所有层按顺序组合
        self.network = nn.Sequential(*layers)
    
    def forward(self, state):
        """
        前向传播
        
        参数:
            state: 输入状态张量
        
        返回:
            每个动作的Q值
        """
        return self.network(state)


class ReplayBuffer:
    """
    经验回放缓冲区
    
    为什么需要经验回放？
    1. 打破数据相关性：连续的交易数据高度相关，
       随机采样可以获得更独立的样本
    2. 提高数据效率：同一经验可以被多次使用
    3. 稳定训练：避免训练过程的剧烈波动
    """
    
    def __init__(self, capacity: int = 100000):
        """
        初始化缓冲区
        
        参数:
            capacity: 最大容量，超出后自动删除最旧的经验
        """
        # deque: 双端队列，自动维护最大长度
        self.buffer = deque(maxlen=capacity)
    
    def push(self, state, action, reward, next_state, done):
        """
        添加一条经验
        """
        self.buffer.append(Experience(state, action, reward, next_state, done))
    
    def sample(self, batch_size: int):
        """
        随机采样一批经验
        
        参数:
            batch_size: 采样数量
        
        返回:
            states, actions, rewards, next_states, dones 张量
        """
        # 随机采样
        experiences = random.sample(self.buffer, batch_size)
        
        # 将列表转换为张量（批量处理）
        states = torch.FloatTensor(np.array([e.state for e in experiences]))
        actions = torch.LongTensor([e.action for e in experiences])
        rewards = torch.FloatTensor([e.reward for e in experiences])
        next_states = torch.FloatTensor(np.array([e.next_state for e in experiences]))
        dones = torch.FloatTensor([e.done for e in experiences])
        
        return states, actions, rewards, next_states, dones
    
    def __len__(self):
        """返回当前缓冲区大小"""
        return len(self.buffer)


class DQNAgent:
    """
    DQN 交易智能体
    
    使用 Double DQN 技术来稳定训练：
    - 主网络 (q_network): 选择动作
    - 目标网络 (target_network): 计算目标Q值
    """
    
    def __init__(self, state_size: int, action_size: int, 
                 learning_rate: float = 0.001,
                 gamma: float = 0.99,
                 epsilon_start: float = 1.0,
                 epsilon_end: float = 0.01,
                 epsilon_decay: float = 0.995,
                 buffer_size: int = 100000,
                 batch_size: int = 64,
                 target_update_freq: int = 10):
        """
        初始化DQN智能体
        
        参数说明:
            state_size: 状态空间维度
            action_size: 动作空间大小
            learning_rate: 学习率
                - 太大：训练不稳定
                - 太小：收敛太慢
                - 推荐：0.001 左右
            gamma: 折扣因子 (0-1)
                - 接近1：更重视长期收益
                - 接近0：更重视短期收益
                - 推荐：0.99
            epsilon_start: 初始探索率
                - 训练初期随机探索的概率
            epsilon_end: 最终探索率
                - 训练后期保留的最小探索概率
            epsilon_decay: 探索率衰减系数
                - 每步之后 epsilon = epsilon * decay
            buffer_size: 经验缓冲区大小
            batch_size: 每次训练的样本数量
            target_update_freq: 目标网络更新频率
        """
        self.state_size = state_size
        self.action_size = action_size
        self.gamma = gamma
        self.epsilon = epsilon_start
        self.epsilon_end = epsilon_end
        self.epsilon_decay = epsilon_decay
        self.batch_size = batch_size
        self.target_update_freq = target_update_freq
        
        # 选择设备：GPU（如果可用）或 CPU
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        print(f"Using device: {self.device}")
        
        # ============================================
        # 创建神经网络
        # ============================================
        # 主网络：用于选择动作和训练
        self.q_network = QNetwork(state_size, action_size).to(self.device)
        
        # 目标网络：用于计算目标Q值
        # 初始时与主网络相同
        self.target_network = QNetwork(state_size, action_size).to(self.device)
        self.target_network.load_state_dict(self.q_network.state_dict())
        
        # Adam优化器：自适应学习率优化算法
        self.optimizer = optim.Adam(self.q_network.parameters(), lr=learning_rate)
        
        # MSE损失函数：均方误差
        self.loss_fn = nn.MSELoss()
        
        # 经验回放缓冲区
        self.memory = ReplayBuffer(buffer_size)
        
        self.training_step = 0
        self.losses = []  # 记录损失值
    
    def select_action(self, state: np.ndarray, training: bool = True) -> int:
        """
        选择动作 (epsilon-greedy策略)
        
        策略说明：
        - 以概率 epsilon 随机选择动作（探索）
        - 以概率 1-epsilon 选择Q值最大的动作（利用）
        
        为什么需要探索？
        - 如果总是选择当前最优，可能错过更好的策略
        - 探索可以发现新的盈利机会
        
        参数:
            state: 当前状态
            training: 是否为训练模式
        
        返回:
            选择的动作 (0, 1, 或 2)
        """
        # 训练时：epsilon-greedy
        if training and random.random() < self.epsilon:
            # 随机探索
            return random.randint(0, self.action_size - 1)
        
        # 评估时或利用时：选择最优动作
        with torch.no_grad():  # 不计算梯度，节省内存
            # 将状态转换为张量
            state_tensor = torch.FloatTensor(state).unsqueeze(0).to(self.device)
            # 计算Q值
            q_values = self.q_network(state_tensor)
            # 返回Q值最大的动作
            return q_values.argmax().item()
    
    def store_experience(self, state, action, reward, next_state, done):
        """存储一条经验到缓冲区"""
        self.memory.push(state, action, reward, next_state, done)
    
    def train_step(self) -> float:
        """
        执行一步训练
        
        训练流程：
        1. 从缓冲区采样一批经验
        2. 计算当前Q值
        3. 计算目标Q值（使用Double DQN）
        4. 计算损失
        5. 反向传播更新网络
        
        返回:
            训练损失值
        """
        # 缓冲区样本不足时不训练
        if len(self.memory) < self.batch_size:
            return 0.0
        
        # ============================================
        # 1. 采样经验
        # ============================================
        states, actions, rewards, next_states, dones = self.memory.sample(self.batch_size)
        states = states.to(self.device)
        actions = actions.to(self.device)
        rewards = rewards.to(self.device)
        next_states = next_states.to(self.device)
        dones = dones.to(self.device)
        
        # ============================================
        # 2. 计算当前Q值
        # ============================================
        # q_network(states) 输出 shape: (batch_size, action_size)
        # gather(1, actions.unsqueeze(1)) 选择执行的动作对应的Q值
        current_q_values = self.q_network(states).gather(1, actions.unsqueeze(1)).squeeze()
        
        # ============================================
        # 3. 计算目标Q值 (Double DQN)
        # ============================================
        # Double DQN 的关键改进：
        # - 标准DQN: 用同一个网络选择动作和评估Q值
        # - Double DQN: 用主网络选择动作，用目标网络评估Q值
        # 这样可以减少Q值的过估计问题
        
        with torch.no_grad():  # 目标值不需要梯度
            # 用主网络选择最优动作
            next_actions = self.q_network(next_states).argmax(1)
            # 用目标网络评估Q值
            next_q_values = self.target_network(next_states).gather(
                1, next_actions.unsqueeze(1)
            ).squeeze()
            # 贝尔曼方程：Q = r + γ * max Q(s', a')
            # done=1 时，没有下一状态，所以 (1-done)*next_q = 0
            target_q_values = rewards + (1 - dones) * self.gamma * next_q_values
        
        # ============================================
        # 4. 计算损失
        # ============================================
        loss = self.loss_fn(current_q_values, target_q_values)
        
        # ============================================
        # 5. 反向传播
        # ============================================
        self.optimizer.zero_grad()  # 清除之前的梯度
        loss.backward()             # 计算梯度
        
        # 梯度裁剪：防止梯度爆炸
        torch.nn.utils.clip_grad_norm_(self.q_network.parameters(), 1.0)
        
        self.optimizer.step()       # 更新权重
        
        self.training_step += 1
        
        # ============================================
        # 6. 更新目标网络
        # ============================================
        # 每隔一定步数，将主网络的参数复制到目标网络
        # 这样目标网络会"落后"于主网络，提供稳定的学习目标
        if self.training_step % self.target_update_freq == 0:
            self.target_network.load_state_dict(self.q_network.state_dict())
        
        # ============================================
        # 7. 衰减探索率
        # ============================================
        # 随着训练进行，逐渐减少探索，增加利用
        self.epsilon = max(self.epsilon_end, self.epsilon * self.epsilon_decay)
        
        loss_value = loss.item()
        self.losses.append(loss_value)
        
        return loss_value
    
    def save(self, filepath: str):
        """
        保存模型
        
        保存内容：
        - 主网络参数
        - 目标网络参数
        - 优化器状态
        - 当前探索率
        - 训练步数
        """
        torch.save({
            'q_network_state_dict': self.q_network.state_dict(),
            'target_network_state_dict': self.target_network.state_dict(),
            'optimizer_state_dict': self.optimizer.state_dict(),
            'epsilon': self.epsilon,
            'training_step': self.training_step
        }, filepath)
        print(f"Model saved to {filepath}")
    
    def load(self, filepath: str):
        """
        加载模型
        """
        checkpoint = torch.load(filepath, map_location=self.device)
        self.q_network.load_state_dict(checkpoint['q_network_state_dict'])
        self.target_network.load_state_dict(checkpoint['target_network_state_dict'])
        self.optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
        self.epsilon = checkpoint['epsilon']
        self.training_step = checkpoint['training_step']
        print(f"Model loaded from {filepath}")
```
