### 1.1 用户注册流程

```plantuml
@startuml
title 用户注册流程

start
:访问注册页面;

partition 选择注册方式 {
  if (注册方式?) then (邮箱注册)
    :输入邮箱地址;
    if (邮箱格式有效?) then (是)
      if (邮箱已注册?) then (是)
        :显示"邮箱已注册"错误;
        :提供"去登录"链接;
        stop
      else (否)
        :点击"获取验证码";
        :发送邮箱验证码;
      endif
    else (否)
      :显示"邮箱格式错误"提示;
      stop
    endif
  else (手机号注册)
    :输入手机号;
    if (手机号格式有效?) then (是)
      if (手机号已注册?) then (是)
        :显示"手机号已注册"错误;
        stop
      else (否)
        :点击"获取验证码";
        :发送短信验证码;
      endif
    else (否)
      :显示"手机号格式错误"提示;
      stop
    endif
  endif
}

partition 验证码验证 {
  :输入6位验证码;
  if (验证码正确且未过期?) then (是)
    :验证通过;
  else (否)
    :显示"验证码错误/已过期";
    if (重试次数<3?) then (是)
      :可重新输入;
    else (否)
      :账户锁定10分钟;
      stop
    endif
  endif
}

partition 设置密码 {
  :输入密码;
  :实时检测密码强度;
  if (密码符合规则?) then (是)
    :显示强度指示器;
  else (否)
    :显示密码规则提示;
    stop
  endif
  
  :输入确认密码;
  if (两次密码一致?) then (是)
    :密码验证通过;
  else (否)
    :显示"密码不一致"错误;
    stop
  endif
}

partition 用户协议 {
  if (已勾选用户协议?) then (是)
    :点击"注册"按钮;
    :按钮显示loading状态;
  else (否)
    :注册按钮禁用;
    :提示"请先阅读并同意用户协议";
    stop
  endif
}

partition 注册完成 {
  if (注册成功?) then (是)
    :显示"注册成功"Toast;
    :自动登录;
    :跳转到首页;
  else (否)
    :显示错误原因;
    :提供重试;
  endif
}

stop
@enduml
```

    :显示"仅只读权限"警告;
    if (用户确认继续?) then (是)
      :标记为"只读";
    else (否)
      stop
    endif
  endif

  :AES-256加密存储;
  :添加到配置列表;
  :关闭弹窗;
  :显示"添加成功"Toast;

### 1.4 交易下单流程

```plantuml
@startuml
title 交易下单流程（市价/限价）

start
:选择交易对;
:加载交易面板;

partition 选择交易方向 {
  if (交易方向?) then (买入)
    :显示买入面板(绿色主题);
    :显示USDT可用余额;
  else (卖出)
    :显示卖出面板(红色主题);
    :显示币种可用持仓;
  endif
}

partition 选择订单类型 {
  if (订单类型?) then (市价单)
    :价格输入框禁用;
    :显示"以市场最优价成交";
    
    if (买入?) then (是)
      :输入USDT金额;
    else (否)
      :输入卖出数量;
    endif
    
    :实时计算预估成交量;
    
  else (限价单)
    :价格输入框启用;
    :默认填入当前价格;
    
    :输入限价;
    if (限价偏离>10%?) then (是)
      :显示价格偏离警告;
    endif
    
    :输入数量;
    :实时计算订单金额;
  endif
}

partition 金额/数量验证 {
  if (余额/持仓充足?) then (是)
    :启用下单按钮;
  else (否)
    :显示"余额不足"/"持仓不足";
    :禁用下单按钮;
    stop
  endif
  
  if (>=最小交易额?) then (是)
    :验证通过;
  else (否)
    :显示"最小交易金额10 USDT";
    stop
  endif
}

partition 订单确认 {
  :点击"买入/卖出"按钮;
  :弹出确认弹窗;
  :显示订单详情;
  
  if (用户确认?) then (是)
    :按钮显示loading;
    :提交订单到交易所;
  else (否)
    :关闭弹窗;
    stop
  endif
}

partition 订单执行 {
  if (提交成功?) then (是)
    if (市价单?) then (是)
      :等待成交(<3s);
      :显示成交结果;
      :更新余额和持仓;
    else (否)
      :添加到当前挂单列表;
      :冻结相应资金/持仓;
    endif
    :显示"下单成功"Toast;
  else (否)
    :显示错误原因;
    :提供重试;
  endif
}

stop
@enduml
```

### 1.6 资产同步与查看流程

```plantuml
@startuml
title 资产同步与查看流程

start
:进入资产页面;

partition 自动同步 {
  fork
    :后台定时同步(30s);
    :调用交易所API获取余额;
    
    if (同步成功?) then (是)
      :更新本地余额数据;
      :更新"最后同步时间";
    else (否)
      :重试(最多3次);
      if (连续失败3次?) then (是)
        :发送告警通知;
        :显示"同步失败"状态;
      endif
    endif
  fork again
    :用户浏览资产页面;
  end fork
}

partition 手动刷新 {
  :点击"刷新"按钮;
  :按钮显示loading;
  :立即触发同步;
  
  if (同步成功?) then (是)
    :更新余额显示;
    :显示"已同步"状态;
    :更新同步时间;
  else (否)
    :显示"同步失败";
    :提供"重试"按钮;
  endif
}

partition 资产展示 {
  :显示总资产(折合USDT);
  :显示今日盈亏;
  
  :加载资产分布饼图;
  :加载7天趋势折线图;
  
  :加载持仓明细表格;
  note right
    币种、可用、冻结
    当前价、估值、盈亏
  end note
}

stop
@enduml
```

| 已完成 | 修改配置 | 已配置 | 清除上次结果 |
| 已完成 | 修改策略 | 已配置 | 清除上次结果 |
| 失败 | 修改策略 | 未配置 | 定位错误行 |

## 第6章：防抖节流与超时设置

### 6.1 防抖设置（Debounce）

| 交互类型 | 防抖时间 | 应用场景 |
|---------|---------|---------|
| 输入框输入 | 200ms | 搜索框、筛选输入、表单校验 |
| 金额/数量输入 | 200ms | 下单金额、回测资金 |
| 表单失焦校验 | 200ms | 邮箱、手机号、密码校验 |
| 按钮点击 | 300ms | 登录、注册、下单、撤单等 |

### 6.2 节流设置（Throttle）

| 交互类型 | 节流时间 | 应用场景 |
|---------|---------|---------|
| 图表滚轮缩放 | 100ms | K线图、深度图缩放 |
| 图表拖拽 | 100ms | K线图拖拽查看历史 |
| 窗口resize | 200ms | 响应式布局调整 |
| 页面滚动 | 100ms | 懒加载、无限滚动 |

### 6.3 超时设置

| 请求类型 | 超时时间 | 超时处理 |
|---------|---------|---------|
| 普通API请求 | 10秒 | 显示"请求超时，请重试" |
| 验证码发送 | 10秒 | 显示"发送超时，请重试" |
| API验证请求 | 10秒 | 显示"验证超时，请稍后重试" |
| 下单请求 | 5秒 | 显示"下单超时，请检查订单状态" |
| 文件上传 | 30秒 | 显示"上传超时" |
| WebSocket重连 | 3秒 | 自动重试，最多3次 |

### 6.4 自动保存设置

| 场景 | 保存间隔 | 保存内容 |
|------|---------|---------|
| 策略代码编辑 | 30秒 | 自动保存到本地草稿 |
| 表单填写 | 失焦时 | 保存表单数据到sessionStorage |
| 回测配置 | 即时 | 保存配置到localStorage |
