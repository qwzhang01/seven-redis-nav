快速示例

Python 
from cryptoquant import Strategy, Context

class MACrossStrategy(Strategy):
    """双均线交叉策略"""
    
    def __init__(self):
        self.short_period = 5
        self.long_period = 20
    
    def on_bar(self, ctx: Context):
        # 计算均线
        ma_short = ctx.ta.SMA(ctx.close, self.short_period)
        ma_long = ctx.ta.SMA(ctx.close, self.long_period)
        
        # 金叉买入
        if ma_short[-2] < ma_long[-2] and ma_short[-1] > ma_long[-1]:
            ctx.buy(percent=100)
        
        # 死叉卖出
        if ma_short[-2] > ma_long[-2] and ma_short[-1] < ma_long[-1]:
            ctx.sell(percent=100)

# 运行回测
if __name__ == "__main__":
    strategy = MACrossStrategy()
    strategy.backtest(
        symbol="BTC/USDT",
        start="2024-01-01",
        end="2024-12-31",
        initial_capital=10000
    )
